class UserController extends Controller
{
    protected function getTranslation()
    {
        // {{ site.data.languages[page.language]['i18n_2'] }}
        $language = $this
            ->request
            ->getBestLanguage();

        // {{ site.data.languages[page.language]['i18n_3'] }}
        $fileName = 'app/messages/'
                  . $language
                  . '.php';
        if (file_exists($fileName) {
            require $fileName;
        } else {
            // {{ site.data.languages[page.language]['i18n_4'] }}
            require 'app/messages/en.php';
        }

        // {{ site.data.languages[page.language]['i18n_5'] }}
        return new NativeArray(
            array(
                'content' => $messages
            )
        );
    }

    public function indexAction()
    {
        $this->view->name = 'Mike';

        $this->view->t = $this
            ->getTranslation();
    }
}
